CREATE SCHEMA `silvifor` ;


CREATE TABLE `silvifor`.`employees` (
  `idEmployees` INT NOT NULL,
  `FirstName` VARCHAR(100) NULL,
  `LastName` VARCHAR(100) NULL,
  `Certificate` VARCHAR(45) NULL,
  `Enabled` TINYINT NULL,
  `IsLead` TINYINT NULL,
  PRIMARY KEY (`idEmployees`));


CREATE TABLE `silvifor`.`clients` (
  `idClients` INT NOT NULL,
  `Name` VARCHAR(100) NULL,
  PRIMARY KEY (`idClients`));

CREATE TABLE `silvifor`.`requestkeys` (
  `idRequestKeys` INT NOT NULL,
  `RK` VARCHAR(45) NULL,
  `Species` VARCHAR(45) NULL,
  PRIMARY KEY (`idRequestKeys`));

CREATE TABLE `silvifor`.`surveyblocks` (
  `idSurveyBlocks` INT NOT NULL,
  `Client` VARCHAR(45) NULL,
  `TaskName` VARCHAR(100) NULL,
  `BlockName` VARCHAR(45) NULL,
  `BlockHa` DECIMAL NULL,
  `Enabled` TINYINT NULL,
  `ActiveYear` DATETIME NULL,
  PRIMARY KEY (`idSurveyBlocks`));


CREATE TABLE `silvifor`.`tasks` (
  `idtasks` INT NOT NULL,
  `TaskName` VARCHAR(100) NULL,
  PRIMARY KEY (`idtasks`));

CREATE TABLE `silvifor`.`opsrecords` (
  `idOpsRecords` INT NOT NULL,
  `Date` Date,
  `Client` VARCHAR(45) NULL,
  `NameCertificate` VARCHAR(100) NULL,
  `Location` VARCHAR(45) NULL,
  `PestControlled` VARCHAR(100) NULL,
  `Product` VARCHAR(45) NULL,
  `Litres` DECIMAL NULL,
  `Adjuvant` VARCHAR(45) NULL,
  `Alitres` DECIMAL NULL,
  `Area` DECIMAL NULL,
  `Rate` DECIMAL NULL,
  `Method` VARCHAR(100) NULL,
  `WindSpeedAM` DECIMAL NULL,
  `WindDirAM` VARCHAR(45) NULL,
  `HumidityAM` DECIMAL NULL,
  `WindSpeedPM` DECIMAL NULL,
  `WindDirPM` VARCHAR(45) NULL,
  `HumidityPM` DECIMAL NULL,
  PRIMARY KEY (`idOpsRecords`));

  CREATE TABLE `silvifor`.`timesheet` (
  `idTimesheet` INT NOT NULL AUTO_INCREMENT,
  `Date` Date,
  `Client` VARCHAR(45) NULL,
  `BlockName` VARCHAR(45) NULL,
  `BlockHa` DECIMAL NULL,
  `TaskName` VARCHAR(100) NULL,
  `Name` VARCHAR(100) NULL,
  `Hours` DECIMAL NULL,
  `OT1` DECIMAL NULL,
  `OT2` DECIMAL NULL,
  `Ha` DECIMAL NULL,
  `ProRate` DECIMAL NULL,
  `AdjHa` DECIMAL NULL,
  `HaRate` DECIMAL NULL,
  `Species` VARCHAR(45) NULL,
  `Trees` DECIMAL NULL,
  `Fert` DECIMAL NULL,
  `Cones` DECIMAL NULL,
  `$Hr` DECIMAL NULL,
  `$Unit` DECIMAL NULL,
  `CrewLead` VARCHAR(100) NULL,
  `Trucks` INT NULL,
  `DailyKm` DECIMAL NULL,
  `RemHa` DECIMAL NULL,
  `Comments` VARCHAR(1000) NULL,
  PRIMARY KEY (`idTimesheet`));
